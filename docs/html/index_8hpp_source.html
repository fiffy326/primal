44template <typename T>
45requires std::is_unsigned_v<T>
46void index(T n) {
47 // Prime number theorem estimate of the Nth prime.
48 T ceiling_estimate = n * std::log(n) + n * std::log(std::log(n));
49
50 // Minimum sieve ceiling (in case the estimate is below the Nth prime).
51 T ceiling_min = 15;
52
53 // Actual sieve ceiling used for computations.
54 T ceiling;
55
56 // Prepare a vector to store the sieved primes.
57 std::vector<T> primes;
58
59 // Sieve until the Nth prime is found.
60 while (true) {
61 // (Re)compute the sieve ceiling.
62 ceiling = std::max(ceiling_estimate, ceiling_min);
63
64 // Compute the primes up to the ceiling.
65 utils::math::sieve(ceiling, primes);
66
67 // Subtract 1 to compensate for zero-based indexing.
68 T i = n - 1;
69
70 // Display the Nth prime if it was found.
71 if (i < primes.size()) {
72 std::cout << "Prime #" << n << " = " << primes[i] << "\n";
73 break;
74 }
75
76 // Increase the sieve ceiling by 10% if the Nth prime wasn't found.
77 std::cout << "Increasing sieve ceiling by 10%...\n";
78 ceiling_min = static_cast<T>(ceiling * 1.1);
79
80 // Clear the primes vector before sieving again.
81 primes.clear();
82 }
83}
