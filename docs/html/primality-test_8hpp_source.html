66template <typename T>
67std::optional<Primality> preliminaryCheck(T number) {
68 if ((number == 0) || (number == 1)) return Primality::NEITHER;
69 if ((number == 2) || (number == 3)) return Primality::PRIME;
70 if (!(number % 2) || !(number % 3)) return Primality::COMPOSITE;
71 return std::nullopt;
72}

80template <typename T>
81requires std::is_unsigned_v<T>
82Primality divisionTest(T number) {
83 // Filter out easy-to-categorize numbers.
84 if (auto result = preliminaryCheck(number)) return *result;
85
86 // Test remaining numbers using trial division.
87 for (T i = 5; i * i < number; i += 6) {
88 if (!(number % i) || !(number % (i + 2))) return Primality::COMPOSITE;
89 }
90
91 // The number must be prime if it survived the trial division test.
92 return Primality::PRIME;
93}

101template <typename T>
102requires std::is_unsigned_v<T>
103Primality sieveTest(T number) {
104 // Filter out easy-to-categorize numbers.
105 if (auto result = preliminaryCheck(number)) return *result;
106
107 // Test remaining numbers using a Sieve of Eratosthenes.
108 std::vector<T> primes;
109
110 // If the final element in the vector is the number, then it is prime.
111 if (!primes.empty() && primes.back() == number) return Primality::PRIME;
112
113 // The number must be composite if the Sieve didn't generate it.
114 return Primality::COMPOSITE;
115}

123template <typename T>
124requires std::is_unsigned_v<T>
125bool isPrime(T number) {
126 return sieveTest(number) == Primality::PRIME;
127}
