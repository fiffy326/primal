47template <typename T>
48std::optional<Primality> preliminaryCheck(T n) {
49 if ((n == 0) || (n == 1)) return Primality::NEITHER;
50 if ((n == 2) || (n == 3)) return Primality::PRIME;
51 if (!(n % 2) || !(n % 3)) return Primality::COMPOSITE;
52 return std::nullopt;
53}

61template <typename T>
62requires std::is_unsigned_v<T>
63Primality divisionTest(T n) {
64 // Perform a preliminary check to filter out easy-to-categorize numbers.
65 if (auto result = preliminaryCheck(n)) return *result;
66
67 // Test the surviving numbers using trial division.
68 for (T i = 5; i * i < n; i += 6) {
69 if (!(n % i) || !(n % (i + 2))) return Primality::COMPOSITE;
70 }
71
72 // Any number that survives the trial division test must be prime.
73 return Primality::PRIME;
74}

82template <typename T>
83requires std::is_unsigned_v<T>
84Primality sieveTest(T n) {
85 // Perform a preliminary check to filter out easy-to-categorize numbers.
86 if (auto result = preliminaryCheck(n)) return *result;
87
88 // Test the surviving numbers using a Sieve of Eratosthenes.
89 std::vector<T> primes;
90 sieve(n, primes);
91
92 // If the number is prime, it will be the last element in the vector.
93 if (!primes.empty() && primes.back() == n) return Primality::PRIME;
94
95 // If the sieve did not produce the number, it is composite.
96 return Primality::COMPOSITE;
97}

105template <typename T>
106requires std::is_unsigned_v<T>
107bool isPrime(T n) {
108 return sieveTest(n) == Primality::PRIME;
